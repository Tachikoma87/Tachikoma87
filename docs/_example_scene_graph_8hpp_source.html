36 void init(void) override{
37 initWindowAndRenderDevice();
38 initCameraAndLights();
39
40 initSkybox();
41 initFPSLabel();
42
43 m_RootSGN.init(nullptr);
44 m_SG.rootNode(&m_RootSGN);
45
46 // initialize ground plane
47 T3DMesh<float> M;
48
49 // load the ground model
50 //SAssetIO::load("Assets/ExampleScenes/TexturedGround.gltf", &M);
51 PrimitiveShapeFactory::plane(&M, Vector2f(1250.0f, 1250.0f), Vector2i(10, 10));
52 setMeshShader(&M, 0.6f, 0.2f);
53 M.changeUVTiling(Vector3f(250.0f, 250.0f, 1.0f));
54 M.computePerVertexNormals();
55 M.computePerVertexTangents();
56 M.getMaterial(0)->TexAlbedo = "Assets/ExampleScenes/Textures/Ground003/Ground003_2K_Color.webp";
57 M.getMaterial(0)->TexNormal = "Assets/ExampleScenes/Textures/Ground003/Ground003_2K_NormalGL.webp";
58 m_Ground.init(&M);
59 BoundingVolume BV;
60 m_Ground.boundingVolume(BV);
61 M.clear();
62
63 // initialize ground transformation and geometry scene graph node
64 m_GroundTransformSGN.init(&m_RootSGN);
65 m_GroundSGN.init(&m_GroundTransformSGN, &m_Ground);
66
67 // load the tree models
68 SAssetIO::load("Assets/ExampleScenes/Trees/LowPolyTree_01.gltf", &M);
69 setMeshShader(&M, 0.8f, 0.04f);
70 M.computePerVertexNormals();
71 scaleAndOffsetModel(&M, 0.5f);
72 M.computeAxisAlignedBoundingBox();
73 m_Trees[0].init(&M);
74 M.clear();
75
76 SAssetIO::load("Assets/ExampleScenes/Trees/LowPolyTree_02.gltf", &M);
77 setMeshShader(&M, 0.8f, 0.04f);
78 M.computePerVertexNormals();
79 M.computeAxisAlignedBoundingBox();
80 m_Trees[1].init(&M);
81 M.clear();
82
83 SAssetIO::load("Assets/ExampleScenes/Trees/LowPolyTree_03.gltf", &M);
84 setMeshShader(&M, 0.8f, 0.04f);
85 M.computePerVertexNormals();
86 scaleAndOffsetModel(&M, 5.0f, Vector3f(0.0f, 0.25f, 0.0f));
87 M.computeAxisAlignedBoundingBox();
88 m_Trees[2].init(&M);
89 M.clear();
90
91 // sceen graph node that holds our forest
92 m_TreeGroupSGN.init(&m_RootSGN);
93
94 float Area = 500.0f; // square area [-Area, Area] on the xz-plane, where trees are planted
95 float TreeCount = 900; // number of trees to create
96
97 for (uint32_t i = 0; i < TreeCount; ++i) {
98 // create the scene graph nodes
99 SGNGeometry* pGeomSGN = nullptr;
100 SGNTransformation* pTransformSGN = nullptr;
101
102 // initialize position and scaling of the tree
103 pTransformSGN = new SGNTransformation();
104 pTransformSGN->init(&m_TreeGroupSGN);
105
106 float TreeScale = CForgeMath::randRange(0.1f, 3.0f);
107
108 Vector3f TreePos = Vector3f::Zero();
109 TreePos.x() = CForgeMath::randRange(-Area, Area);
110 TreePos.z() = CForgeMath::randRange(-Area, Area);
111
112 pTransformSGN->translation(TreePos);
113 pTransformSGN->scale(Vector3f(TreeScale, TreeScale, TreeScale));
114
115 // initialize geometry
116 // choose one of the trees randomly
117 pGeomSGN = new SGNGeometry();
118 uint8_t TreeType = CForgeMath::rand() % 3;
119 pGeomSGN->init(pTransformSGN, &m_Trees[TreeType]);
120
121 m_TreeTransformSGNs.push_back(pTransformSGN);
122 m_TreeSGNs.push_back(pGeomSGN);
123
124 }//for[TreeCount]
125
126 // change sun settings to cover this large area
127 m_Sun.position(Vector3f(100.0f, 1000.0f, 500.0f));
128 m_Sun.initShadowCasting(2048*2, 2048*2, Vector2i(1000, 1000), 1.0f, 5000.0f);
129
130 m_Fly = false;
131 }//initialize
132
133 void clear(void) override{
134 for (auto& i : m_TreeSGNs) if (nullptr != i) delete i;
135 for (auto& i : m_TreeTransformSGNs) if (nullptr != i) delete i;
136
137 ExampleSceneBase::clear();
138 }//clear
139
140 void mainLoop(void) override{
141
142 m_RenderWin.update();
143
144 defaultCameraUpdate(&m_Cam, m_RenderWin.keyboard(), m_RenderWin.mouse(), 0.1f * 60.0f/m_FPS, 1.0f, 8.0f);
145 // make sure to always walk on the ground if not flying
146 if (!m_Fly) {
147 Vector3f CamPos = m_Cam.position();
148 CamPos.y() = 1.0f;
149 m_Cam.position(CamPos);
150 }
151
152 m_SkyboxSG.update(60.0f/m_FPS);
153 m_SG.update(60.0f / m_FPS);
154
155 m_RenderDev.activePass(RenderDevice::RENDERPASS_SHADOW, &m_Sun);
156 m_RenderDev.activeCamera(const_cast<VirtualCamera*>(m_Sun.camera()));
157 m_SG.render(&m_RenderDev);
158
159 m_RenderDev.activePass(RenderDevice::RENDERPASS_GEOMETRY);
160 m_RenderDev.activeCamera(&m_Cam);
161 m_SG.render(&m_RenderDev);
162
163 m_RenderDev.activePass(RenderDevice::RENDERPASS_LIGHTING);
164
165 m_RenderDev.activePass(RenderDevice::RENDERPASS_FORWARD, nullptr, false);
166 m_SkyboxSG.render(&m_RenderDev);
167 m_FPSLabel.render(&m_RenderDev);
168
169 m_RenderWin.swapBuffers();
170
171 updateFPS();
172
173 // change between flying and walking mode
174 if (m_RenderWin.keyboard()->keyPressed(Keyboard::KEY_F, true)) m_Fly = !m_Fly;
175
176 defaultKeyboardUpdate(m_RenderWin.keyboard());
177
178 }//run

182 void scaleAndOffsetModel(T3DMesh<float>* pModel, float Factor, Vector3f Offset = Vector3f::Zero()) {
183 Matrix3f Sc = Matrix3f::Identity();
184 Sc(0, 0) = Factor;
185 Sc(1, 1) = Factor;
186 Sc(2, 2) = Factor;
187 for (uint32_t i = 0; i < pModel->vertexCount(); ++i) pModel->vertex(i) = Sc * pModel->vertex(i) - Offset;
188 }//scaleModel
