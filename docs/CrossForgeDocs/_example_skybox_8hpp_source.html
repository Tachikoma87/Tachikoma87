123 void mainLoop(void)override {
124 m_RenderWin.update();
125 m_SG.update(60.0f / m_FPS);
126 m_SkyboxSG.update(60.0f / m_FPS);
127
128 // handle input for the skybox
129 Keyboard* pKeyboard = m_RenderWin.keyboard();
130 float Step = (pKeyboard->keyPressed(Keyboard::KEY_LEFT_SHIFT)) ? -0.05f : 0.05f;
131 if (pKeyboard->keyPressed(Keyboard::KEY_1, true)) m_Skybox.brightness(m_Skybox.brightness() + Step);
132 if (pKeyboard->keyPressed(Keyboard::KEY_2, true)) m_Skybox.saturation(m_Skybox.saturation() + Step);
133 if (pKeyboard->keyPressed(Keyboard::KEY_3, true)) m_Skybox.contrast(m_Skybox.contrast() + Step);
134 if (pKeyboard->keyPressed(Keyboard::KEY_LEFT_SHIFT) && pKeyboard->keyPressed(Keyboard::KEY_R)) {
135 m_SkyboxTransSGN.rotationDelta(Quaternionf::Identity());
136 }
137 else if (pKeyboard->keyPressed(Keyboard::KEY_R, true)) {
138 Quaternionf RDelta;
139 RDelta = AngleAxisf(CForgeMath::degToRad(-2.5f / 60.0f), Vector3f::UnitY());
140 m_SkyboxTransSGN.rotationDelta(RDelta);
141 }
142
143 if (pKeyboard->keyPressed(Keyboard::KEY_F1, true)) m_Skybox.init(m_ClearSky[0], m_ClearSky[1], m_ClearSky[2], m_ClearSky[3], m_ClearSky[4], m_ClearSky[5]);
144 if (pKeyboard->keyPressed(Keyboard::KEY_F2, true)) m_Skybox.init(m_EmptySpace[0], m_EmptySpace[1], m_EmptySpace[2], m_EmptySpace[3], m_EmptySpace[4], m_EmptySpace[5]);
145 if (pKeyboard->keyPressed(Keyboard::KEY_F3, true)) m_Skybox.init(m_Techno[0], m_Techno[1], m_Techno[2], m_Techno[3], m_Techno[4], m_Techno[5]);
146 if (pKeyboard->keyPressed(Keyboard::KEY_F4, true)) m_Skybox.init(m_BlueCloud[0], m_BlueCloud[1], m_BlueCloud[2], m_BlueCloud[3], m_BlueCloud[4], m_BlueCloud[5]);
147
148 defaultCameraUpdate(&m_Cam, m_RenderWin.keyboard(), m_RenderWin.mouse());
149
150 m_RenderDev.activePass(RenderDevice::RENDERPASS_SHADOW, &m_Sun);
151 m_RenderDev.activeCamera((VirtualCamera*)m_Sun.camera());
152 m_SG.render(&m_RenderDev);
153
154 m_RenderDev.activePass(RenderDevice::RENDERPASS_GEOMETRY);
155 m_RenderDev.activeCamera(&m_Cam);
156 m_SG.render(&m_RenderDev);
157
158 m_RenderDev.activePass(RenderDevice::RENDERPASS_LIGHTING);
159
160 m_RenderDev.activePass(RenderDevice::RENDERPASS_FORWARD, nullptr, false);
161 // Skybox should be last thing to render
162 m_SkyboxSG.render(&m_RenderDev);
163 m_FPSLabel.render(&m_RenderDev);
164
165 m_RenderWin.swapBuffers();
166
167 updateFPS();
168 defaultKeyboardUpdate(m_RenderWin.keyboard());
169 }
