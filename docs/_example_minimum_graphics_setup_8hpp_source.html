41 void init() override{
42
43 initWindowAndRenderDevice();
44 initCameraAndLights();
45 initFPSLabel();
46
47 // build scene graph
48 m_RootSGN.init(nullptr);
49 m_SG.init(&m_RootSGN);
50
51 // load skydome and a textured cube
52 T3DMesh<float> M;
53
54 initGroundPlane(&m_RootSGN, 100.0f, 20.0f);
55
56 SAssetIO::load("Assets/ExampleScenes/Duck/Duck.gltf", &M);
57 for (uint32_t i = 0; i < M.materialCount(); ++i) CForgeUtility::defaultMaterial(M.getMaterial(i), CForgeUtility::PLASTIC_YELLOW);
58 M.computePerVertexNormals();
59 m_Duck.init(&M);
60 M.clear();
61
62
63 // add cube
64 m_DuckTransformSGN.init(&m_RootSGN, Vector3f(0.0f, 1.5f, 0.0f));
65 m_DuckSGN.init(&m_DuckTransformSGN, &m_Duck);
66 m_DuckSGN.scale(Vector3f(0.02f, 0.02f, 0.02f));
67
68 // rotate about the y-axis at 45 degree every second
69 Quaternionf R;
70 R = AngleAxisf(CForgeMath::degToRad(45.0f / 60.0f), Vector3f::UnitY());
71 m_DuckTransformSGN.rotationDelta(R);
72
73 // create help text
74 LineOfText* pKeybindings = new LineOfText();
75 pKeybindings->init(CForgeUtility::defaultFont(CForgeUtility::FONTTYPE_SANSERIF, 18), "Movement: W,A,S,D | Rotation: LMB/RMB + Mouse | F1: Toggle help text");
76 m_HelpTexts.push_back(pKeybindings);
77
78 std::string ErrorMsg;
79 if (0 != CForgeUtility::checkGLError(&ErrorMsg)) {
80 SLogger::log("OpenGL Error" + ErrorMsg, "PrimitiveFactoryTestScene", SLogger::LOGTYPE_ERROR);
81 }
82
83 }//initialize
84
85 void clear(void) override{
86 m_RenderWin.stopListening(this);
87 if (nullptr != m_pShaderMan) m_pShaderMan->release();
88 m_pShaderMan = nullptr;
89 }//clear
90
91
92 void mainLoop(void)override {
93 m_RenderWin.update();
94 m_SG.update(60.0f / m_FPS);
95
96 defaultCameraUpdate(&m_Cam, m_RenderWin.keyboard(), m_RenderWin.mouse());
97
98 m_RenderDev.activePass(RenderDevice::RENDERPASS_SHADOW, &m_Sun);
99 m_RenderDev.activeCamera(const_cast<VirtualCamera*>(m_Sun.camera()));
100 m_SG.render(&m_RenderDev);
101
102 m_RenderDev.activePass(RenderDevice::RENDERPASS_GEOMETRY);
103 m_RenderDev.activeCamera(&m_Cam);
104 m_SG.render(&m_RenderDev);
105
106 m_RenderDev.activePass(RenderDevice::RENDERPASS_LIGHTING);
107 m_FPSLabel.render(&m_RenderDev);
108 if (m_DrawHelpTexts) drawHelpTexts();
109
110 m_RenderWin.swapBuffers();
111
112 updateFPS();
113
114 defaultKeyboardUpdate(m_RenderWin.keyboard());
115
116 }//mainLoop
