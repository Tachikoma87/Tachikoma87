39 void init(void) override{
40
41 initWindowAndRenderDevice();
42 initCameraAndLights();
43
44 // load skydome and a textured cube
45 T3DMesh<float> M;
46
47 SAssetIO::load("Assets/ExampleScenes/SimpleSkydome.gltf", &M);
48 setMeshShader(&M, 0.8f, 0.04f);
49 M.computePerVertexNormals();
50 M.computeAxisAlignedBoundingBox();
51 m_Skydome.init(&M);
52 M.clear();
53
54 SAssetIO::load("Assets/ExampleScenes/Helmet/DamagedHelmet.gltf", &M);
55 setMeshShader(&M, 0.2f, 0.24f);
56 M.computePerVertexNormals();
57 M.computePerVertexTangents();
58 M.computeAxisAlignedBoundingBox();
59 m_Helmet.init(&M);
60 M.clear();
61
62 // build scene graphs (one for every viewport)
63
64 float Speed = 10.0f; // Degree per second
65 Quaternionf HelmetRotationDeltas[4];
66 HelmetRotationDeltas[0] = AngleAxisf(CForgeMath::degToRad(Speed / 60.0f), Vector3f::UnitX());
67 HelmetRotationDeltas[1] = AngleAxisf(CForgeMath::degToRad(Speed / 60.0f), Vector3f::UnitY());
68 HelmetRotationDeltas[2] = AngleAxisf(CForgeMath::degToRad(Speed / 60.0f), Vector3f::UnitZ());
69 HelmetRotationDeltas[3] = AngleAxisf(CForgeMath::degToRad(Speed / 60.0f), Vector3f(1.0f, 1.0f, 1.0f).normalized());
70
71 for (uint8_t i = 0; i < 4; ++i) {
72 m_RootSGNs[i].init(nullptr);
73 m_SGs[i].init(&m_RootSGNs[i]);
74
75 // add skydome
76 m_DomeGeomSGNs[i].init(&m_RootSGNs[i], &m_Skydome, Vector3f::Zero(), Quaternionf::Identity(), Vector3f(50.0f, 50.0f, 50.0f));
77 // add helmet
78 m_HelmetTransSGNs[i].init(&m_RootSGNs[i], Vector3f(0.0f, 3.5f, 0.0f));
79 m_HelmetGeomSGNs[i].init(&m_HelmetTransSGNs[i], &m_Helmet, Vector3f::Zero(), Quaternionf::Identity(), Vector3f(3.0f, 3.0f, 3.0f));
80
81 // let the helmets spin
82 m_HelmetTransSGNs[i].rotationDelta(HelmetRotationDeltas[i]);
83 }
84
85 // we need one viewport for the GBuffer
86 m_GBufferVP.Position = Vector2i(0, 0);
87 m_GBufferVP.Size = Vector2i(m_RenderDev.gBuffer()->width(), m_RenderDev.gBuffer()->height());
88 m_RenderDev.viewport(RenderDevice::RENDERPASS_GEOMETRY, m_GBufferVP);
89
90 updateViewportsAndCamera();
91
92 // stuff for performance monitoring
93 uint64_t LastFPSPrint = CForgeUtility::timestamp();
94 int32_t FPSCount = 0;
95
96 std::string GLError = "";
97 CForgeUtility::checkGLError(&GLError);
98 if (!GLError.empty()) printf("GLError occurred: %s\n", GLError.c_str());
99 }//initialize
100
101 void clear(void) override{
102 ExampleSceneBase::clear();
103 }//clear
104
105 void mainLoop(void) override{
106
107 m_RenderWin.update();
108 defaultCameraUpdate(&m_Cam, m_RenderWin.keyboard(), m_RenderWin.mouse());
109
110 // perform rendering for the 4 viewports
111 for (uint8_t i = 0; i < 4; ++i) {
112 m_SGs[i].update(60.0f / m_FPS);
113
114 // render scene as usual
115 m_RenderDev.activePass(RenderDevice::RENDERPASS_SHADOW, &m_Sun);
116 m_RenderDev.activeCamera((VirtualCamera*)m_Sun.camera());
117 m_SGs[i].render(&m_RenderDev);
118
119 m_RenderDev.activePass(RenderDevice::RENDERPASS_GEOMETRY);
120 m_RenderDev.activeCamera(&m_Cam);
121 m_SGs[i].render(&m_RenderDev);
122
123 // set viewport and perform lighting pass
124 // this will produce the correct tile in the final output window (backbuffer to be specific)
125 m_RenderDev.viewport(RenderDevice::RENDERPASS_LIGHTING, m_VPs[i]);
126 m_RenderDev.activePass(RenderDevice::RENDERPASS_LIGHTING, nullptr, (i == 0) ? true : false);
127 }
128
129 m_RenderWin.swapBuffers();
130
131 updateFPS();
132 defaultKeyboardUpdate(m_RenderWin.keyboard());
133
134 }//mainLoop
135 protected:
136
137 void updateViewportsAndCamera(void) {
138 uint32_t RenderWinWidth = m_RenderWin.width();
139 uint32_t RenderWinHeight = m_RenderWin.height();
140
141 uint32_t Margin = 14;
142 Vector2i VPSize = Vector2i(RenderWinWidth / 2, RenderWinHeight / 2) - 2 * Vector2i(Margin, Margin) + Vector2i(Margin / 2, Margin / 2);
143
144 // Top left
145 m_VPs[0].Position = Vector2i(0, RenderWinHeight / 2) + Vector2i(Margin, Margin / 2);
146 // top right
147 m_VPs[1].Position = Vector2i(RenderWinWidth / 2, RenderWinHeight / 2) + Vector2i(Margin / 2, Margin / 2);
148 // bottom left
149 m_VPs[2].Position = Vector2i(Margin, Margin);
150 // bottom right
151 m_VPs[3].Position = Vector2i(RenderWinWidth / 2, 0) + Vector2i(Margin / 2, Margin);
152 for (uint8_t i = 0; i < 4; ++i) m_VPs[i].Size = VPSize;
153
154 m_Cam.projectionMatrix(m_VPs[0].Size[0], m_VPs[0].Size[1], CForgeMath::degToRad(45.0f), 0.1f, 1000.0f);
155
156 m_GBufferVP.Size = Vector2i(m_RenderDev.gBuffer()->width(), m_RenderDev.gBuffer()->height());
157 m_RenderDev.viewport(RenderDevice::RENDERPASS_GEOMETRY, m_GBufferVP);
158
159 // viewport for forward pass (required for correct screenshots)
160 RenderDevice::Viewport VFPass;
161 VFPass.Position = Vector2i(0, 0);
162 VFPass.Size = Vector2i(RenderWinWidth, RenderWinHeight);
163 m_RenderDev.viewport(RenderDevice::RENDERPASS_FORWARD, VFPass);
164
165 }//updateViewports
166
167 void listen(GLWindowMsg Msg) override{
168 ExampleSceneBase::listen(Msg);
169
170 updateViewportsAndCamera();
171 }
