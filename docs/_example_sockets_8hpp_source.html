46 void init(void) override{
47 initWindowAndRenderDevice();
48 initCameraAndLights();
49
50 T3DMesh<float> M;
51
52 SAssetIO::load("Assets/ExampleScenes/SimpleSkydome.glb", &M);
53 setMeshShader(&M, 0.8f, 0.04f);
54 M.computePerVertexNormals();
55 m_Skydome.init(&M);
56 M.clear();
57
58 SAssetIO::load("Assets/ExampleScenes/Duck/Duck.gltf", &M);
59 setMeshShader(&M, 0.1f, 0.04f);
60 M.computePerVertexNormals();
61 m_Duck.init(&M);
62 M.clear();
63
64 // build scene graph
65 m_RootSGN.init(nullptr);
66 m_SG.init(&m_RootSGN);
67
68 // add skydome
69 m_SkydomeSGN.init(&m_RootSGN, &m_Skydome);
70 m_SkydomeSGN.scale(Vector3f(50.0f, 50.0f, 50.0f));
71
72 // add cube
73 m_DuckTransformSGN.init(&m_RootSGN, Vector3f(0.0f, 3.0f, 0.0f));
74 m_DuckSGN.init(&m_DuckTransformSGN, &m_Duck);
75 m_DuckSGN.scale(Vector3f(0.02f, 0.02f, 0.02f));
76
77 // rotate about the y-axis at 45 degree every second
78 Quaternionf R;
79 R = AngleAxisf(CForgeMath::degToRad(45.0f / 60.0f), Vector3f::UnitY());
80 m_DuckTransformSGN.rotationDelta(R);
81
82 // stuff for performance monitoring
83 uint64_t LastFPSPrint = CForgeUtility::timestamp();
84 int32_t FPSCount = 0;
85
86 std::string GLError = "";
87 CForgeUtility::checkGLError(&GLError);
88 if (!GLError.empty()) printf("GLError occurred: %s\n", GLError.c_str());
89
90 }//initialize

109 void mainLoop(void) override{
110
111 if (m_RenderWin.keyboard()->keyPressed(Keyboard::KEY_1, true)) {
112 try {
113 printf("Starting UDP server...");
114 UDPServer.begin(UDPSocket::TYPE_SERVER, UDPPort);
115 printf("ok\n");
116 printf("Starting UDP Clients...\n\tClient 1...");
117 UDPClient1.begin(UDPSocket::TYPE_CLIENT, 0);
118 printf("ok\n\tClient 2...");
119 UDPClient2.begin(UDPSocket::TYPE_CLIENT, 0);
120 printf("ok\n");
121
122 UDPActive = true;
123 }
124 catch (const CrossForgeException& e) {
125 SLogger::logException(e);
126 }
127
128 }//if[pressed key 1]
129
130 if (UDPActive) {
131 // check server data
132 while (UDPServer.recvData(Buffer, &DataSize, &Sender, &Port)) {
133 printf("Message from %s:%d: %s\n", Sender.c_str(), Port, (char*)Buffer);
134 string Response = "Acknowledged - " + std::string((char*)Buffer);
135 UDPServer.sendData((uint8_t*)Response.c_str(), Response.length() + 1, Sender, Port);
136 }
137
138 if (UDPClient1.recvData(Buffer, &DataSize, &Sender, &Port)) {
139 printf("UDP Client 1 received message from %s:%d: %s\n", Sender.c_str(), Port, (char*)Buffer);
140 }
141 if (UDPClient2.recvData(Buffer, &DataSize, &Sender, &Port)) {
142 printf("UDP Client 2 received message from %s:%d: %s\n", Sender.c_str(), Port, (char*)Buffer);
143 }
144
145
146 if (m_RenderWin.keyboard()->keyPressed(Keyboard::KEY_2, true)) {
147 string Msg = "UDP Client 1: Message " + std::to_string(UDPMsgCounter++);
148 UDPClient1.sendData((uint8_t*)Msg.c_str(), Msg.length() + 1, "127.0.0.1", UDPPort);
149 }
150 if (m_RenderWin.keyboard()->keyPressed(Keyboard::KEY_3, true)) {
151 string Msg = "UDP Client 2: Message " + std::to_string(UDPMsgCounter++);
152 UDPClient2.sendData((uint8_t*)Msg.c_str(), Msg.length() + 1, "127.0.0.1", UDPPort);
153 }
154 }//if[UDPActive]
155
156 // activate servers and clients on demand
157 if (m_RenderWin.keyboard()->keyPressed(Keyboard::KEY_4, true)) {
158 try {
159 printf("Starting TCP server...");
160 TCPServer.begin(TCPSocket::TYPE_SERVER, TCPPort);
161 printf("ok\n");
162 printf("Starting TCP clients: Client 1...");
163 TCPClient1.begin(TCPSocket::TYPE_CLIENT, TCPPort);
164 printf("ok...Client 2...");
165 TCPClient2.begin(TCPSocket::TYPE_CLIENT, TCPPort);
166 printf("ok\n");
167 printf("Connecting clients to server...\n");
168 // connect to server
169 if (TCPClient1.connectTo("127.0.0.1", TCPPort)) printf("\tTCP Client 1 connected to 127.0.0.1:"); printf("%s\n", std::to_string(TCPPort).c_str());
170 if (TCPClient2.connectTo("127.0.0.1", TCPPort)) printf("\tTCP Client 2 connected to 127.0.0.1:"); printf("%s\n", std::to_string(TCPPort).c_str());
171
172 TCPActive = true;
173 }
174 catch (const CrossForgeException& e) {
175 SLogger::logException(e);
176 }
177 }//if[pressed key 1]
178
179 if (TCPActive) {
180 // check data at server
181 for (uint32_t i = 0; i < TCPServer.activeConnections(); ++i) {
182 while (TCPServer.recvData(Buffer, &DataSize, i)) {
183 auto Info = TCPServer.connectionInfo(i);
184 printf("TCP Server received message from %s:%d: %s\n", Info.IP.c_str(), Info.Port, (char*)Buffer);
185 string Response = "Acknowledged - " + std::string((char*)Buffer);;
186 TCPServer.sendData((uint8_t*)Response.c_str(), Response.length() + 1, i);
187 }
188 }
189 // check client 1 receive status
190 if (TCPClient1.recvData(Buffer, &DataSize, 0)) {
191 printf("TCP client 1 received message: %s\n", (char*)Buffer);
192 }
193 if (TCPClient2.recvData(Buffer, &DataSize, 0)) {
194 printf("TCP client 2 received message: %s\n", (char*)Buffer);
195 }
196
197 if (m_RenderWin.keyboard()->keyPressed(Keyboard::KEY_5, true)) {
198 string Msg = "TCP message " + std::to_string(TCPMsgCounter++);
199 TCPClient1.sendData((uint8_t*)Msg.c_str(), Msg.length() + 1, 0);
200 }
201 if (m_RenderWin.keyboard()->keyPressed(Keyboard::KEY_6, true)) {
202 string Msg = "TCP message " + std::to_string(TCPMsgCounter++);
203 TCPClient2.sendData((uint8_t*)Msg.c_str(), Msg.length() + 1, 0);
204 }
205 }//if[TCP active]
206
207 // render something
208 m_RenderWin.update();
209 m_SG.update(m_FPS / 60.0f);
210
211 defaultCameraUpdate(&m_Cam, m_RenderWin.keyboard(), m_RenderWin.mouse());
212
213 m_RenderDev.activePass(RenderDevice::RENDERPASS_SHADOW, &m_Sun);
214 m_SG.render(&m_RenderDev);
215
216 m_RenderDev.activePass(RenderDevice::RENDERPASS_GEOMETRY);
217 m_SG.render(&m_RenderDev);
218
219 m_RenderDev.activePass(RenderDevice::RENDERPASS_LIGHTING);
220
221 m_RenderWin.swapBuffers();
222
223 updateFPS();
224 defaultKeyboardUpdate(m_RenderWin.keyboard());
225
226 }//mainLoop
