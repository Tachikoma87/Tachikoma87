128
129 if (-1 != PlayMTAnimation) {
130 MorphTargetAnimationController::ActiveAnimation* pAnim = m_MTController.play(PlayMTAnimation, MTAnimationSpeed);

154 void buildMTModel(T3DMesh<float>* pBaseMesh) {
155 if (nullptr == pBaseMesh) throw NullpointerExcept("pBaseMesh");
156
157 printf("Building morph target model...");
158 uint64_t Start = CForgeUtility::timestamp();
159
160 // create morph target build and initialize with base mesh
161 MorphTargetModelBuilder MTBuilder;
162 MTBuilder.init(pBaseMesh);
163
164 // define models we want to add
165 // models have to bin in full vertex correspondence (same number of vertices, each having the same meaning)
166 vector<pair<string, string>> MTList;
167 MTList.push_back(pair("Anger", "Assets/ExampleScenes/FaceGenMale/MaleFace_ExpressionAnger.obj"));
168 MTList.push_back(pair("ChinRaised", "Assets/ExampleScenes/FaceGenMale/MaleFace_ExpressionChinRaised.obj"));
169 MTList.push_back(pair("Disgust", "Assets/ExampleScenes/FaceGenMale/MaleFace_ExpressionDisgust.obj"));
170 MTList.push_back(pair("Fear", "Assets/ExampleScenes/FaceGenMale/MaleFace_ExpressionFear.obj"));
171 MTList.push_back(pair("Frown", "Assets/ExampleScenes/FaceGenMale/MaleFace_ExpressionFrown.obj"));
172 MTList.push_back(pair("Kiss", "Assets/ExampleScenes/FaceGenMale/MaleFace_ExpressionKiss.obj"));
173 MTList.push_back(pair("PuffCheeks", "Assets/ExampleScenes/FaceGenMale/MaleFace_ExpressionPuffCheeks.obj"));
174 MTList.push_back(pair("Surprise", "Assets/ExampleScenes/FaceGenMale/MaleFace_ExpressionSurprise.obj"));
175
176
177 // load models and add to builder as targets
178 T3DMesh<float> M;
179 for (auto i : MTList) {
180 M.clear();
181 try {
182 AssetIO::load(i.second, &M);
183 M.computePerVertexNormals();
184 MTBuilder.addTarget(&M, i.first);
185 }
186 catch (const CrossForgeException& e) {
187 SLogger::logException(e);
188 }
189 }//for[all files]
190
191 // build morph targets and retrieve them
192 MTBuilder.build();
193 MTBuilder.retrieveMorphTargets(pBaseMesh);
194
195 printf(" finished in %d ms\n", uint32_t(CForgeUtility::timestamp() - Start));
196 }//buildMTModel
197
198 void buildMTSequences(MorphTargetAnimationController* pController) {
199 if (nullptr == pController) throw NullpointerExcept("pController");
200
201 MorphTargetAnimationController::AnimationSequence Seq;
202
203 // for every morph target we create a sequence
204 // 1.25 seconds to activate the target, 0.5 seconds hold, and 1.25 seconds to go back to base mesh
205 for (uint32_t i = 0; i < pController->morphTargetCount(); ++i) {
206 Seq.clear();
207 MorphTargetAnimationController::MorphTarget* pMT = pController->morphTarget(i);
208 Seq.Name = pMT->Name;
209 Seq.Targets.push_back(pMT->ID);
210 Seq.Targets.push_back(pMT->ID);
211 Seq.Targets.push_back(pMT->ID);
212 Seq.Parameters.push_back(Vector3f(0.0f, 1.0f, 1.25f));
213 Seq.Parameters.push_back(Vector3f(1.0f, 1.0f, 0.5f));
214 Seq.Parameters.push_back(Vector3f(1.0f, 0.0f, 1.25f));
215 pController->addAnimationSequence(&Seq);
216 }//for[all morph targets
217
218
219 }//buildMTSequences
