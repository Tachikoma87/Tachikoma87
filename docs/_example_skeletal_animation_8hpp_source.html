40 void init(void) override{
41
42 initWindowAndRenderDevice();
43 initCameraAndLights();
44
45 // load skydome
46 T3DMesh<float> M;
47
48 SAssetIO::load("Assets/ExampleScenes/SimpleSkydome.gltf", &M);
49 setMeshShader(&M, 0.8f, 0.04f);
50 M.computePerVertexNormals();
51 m_Skydome.init(&M);
52 M.clear();
53
54 // initialize skeletal actor (Eric) and its animation controller
55 SAssetIO::load("Assets/ExampleScenes/CesiumMan/CesiumMan.gltf", &M);
56
57 setMeshShader(&M, 0.7f, 0.04f);
58 M.computePerVertexNormals();
59 m_BipedController.init(&M);
60 m_CesiumMan.init(&M, &m_BipedController);
61 M.clear();
62
63 // build scene graph
64 m_RootSGN.init(nullptr);
65 m_SG.init(&m_RootSGN);
66
67 // add skydome
68 m_SkydomeSGN.init(&m_RootSGN, &m_Skydome);
69 m_SkydomeSGN.scale(Vector3f(50.0f, 50.0f, 50.0f));
70
71 // add skeletal actor to scene graph (Eric)
72 m_CesiumManTransformSGN.init(&m_RootSGN, Vector3f(0.0f, 0.0f, 0.0f));
73 m_CesiumManSGN.init(&m_CesiumManTransformSGN, &m_CesiumMan);
74 m_CesiumManSGN.scale(Vector3f(3.0f, 3.0f, 3.0f));
75
76 Quaternionf Rot;
77 Rot = AngleAxisf(CForgeMath::degToRad(-90.0f), Vector3f::UnitX());
78 m_CesiumManSGN.rotation(Rot);
79
80 // stuff for performance monitoring
81 uint64_t LastFPSPrint = CForgeUtility::timestamp();
82 int32_t FPSCount = 0;
83
84 // check wheter a GL error occurred
85 std::string GLError = "";
86 CForgeUtility::checkGLError(&GLError);
87 if (!GLError.empty()) printf("GLError occurred: %s\n", GLError.c_str());
88
89 m_RepeatAnimation = false;
90 }//initialize
91
92 void clear(void) override{
93 ExampleSceneBase::clear();
94 }
95
96 void mainLoop(void)override {
97 m_RenderWin.update();
98 m_SG.update(60.0f / m_FPS);
99
100 // this will progress all active skeletal animations for this controller
101 m_BipedController.update(60.0f / m_FPS);
102 if (m_RepeatAnimation && nullptr != m_CesiumMan.activeAnimation()) {
103 auto* pAnim = m_CesiumMan.activeAnimation();
104 if (pAnim->t >= pAnim->Duration) pAnim->t -= pAnim->Duration;
105 }
106
107 defaultCameraUpdate(&m_Cam, m_RenderWin.keyboard(), m_RenderWin.mouse());
108
109 // if user hits key 1, animation will be played
110 // if user also presses shift, animation speed is doubled
111 float AnimationSpeed = 1000 / 60.0f;
112 if (m_RenderWin.keyboard()->keyPressed(Keyboard::KEY_LEFT_SHIFT)) AnimationSpeed *= 2.0f;
113 if (m_RenderWin.keyboard()->keyPressed(Keyboard::KEY_LEFT_CONTROL)) AnimationSpeed *= 0.25f;
114 if (m_RenderWin.keyboard()->keyPressed(Keyboard::KEY_1, true)) {
115 SkeletalAnimationController::Animation* pAnim = m_BipedController.createAnimation(0, AnimationSpeed, 0.0f);
116 m_CesiumMan.activeAnimation(pAnim);
117 }
118 if (m_RenderWin.keyboard()->keyPressed(Keyboard::KEY_R, true)) {
119 m_RepeatAnimation = !m_RepeatAnimation;
120 }
121
122 m_RenderDev.activePass(RenderDevice::RENDERPASS_SHADOW, &m_Sun);
123 m_SG.render(&m_RenderDev);
124
125 m_RenderDev.activePass(RenderDevice::RENDERPASS_GEOMETRY);
126 m_SG.render(&m_RenderDev);
127
128 m_RenderDev.activePass(RenderDevice::RENDERPASS_LIGHTING);
129
130 m_RenderWin.swapBuffers();
131
132 updateFPS();
133 defaultKeyboardUpdate(m_RenderWin.keyboard());
134 }//mainLoop
135
136 protected:

145

148 };//ExampleSkeletalAnimation
